CREATE TABLE PROJECT (
PROJ_ID INT AUTO_INCREMENT PRIMARY KEY,
PROJ_NM VARCHAR(50) NOT NULL,
PROJ_MNG VARCHAR(30) NOT NULL 

);

CREATE TABLE WORDDICT (
WORD_ID INT AUTO_INCREMENT PRIMARY KEY,
PROJ_ID INT NOT NULL,
WORD_KOR VARCHAR(50) NOT NULL,
WORD_ENG VARCHAR(50) NOT NULL,
WORD_ABR VARCHAR(50) NOT NULL,
WORD_DEF VARCHAR(50),

FOREIGN KEY (PROJ_ID) REFERENCES PROJECT(PROJ_ID) ON DELETE CASCADE

);
